name: str = "Sudharsan"
age: int = 25
score: float = 9.8

def greet(name: str, formal: bool = False) -> str:
    """Return a greeting string."""
    prefix = "Dear" if formal else "Hey"
    return f"{prefix} {name}!"

try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
finally:
    print("Always runs")
